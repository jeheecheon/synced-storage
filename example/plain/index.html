Cookie

Cookie Store

Persists across sessions. Syncs via cookie.
Open a new tab — the counter carries over.

Usage
import { CookieClient } from "synced-storage/core";

const client = new CookieClient();
const store = client.getOrCreateStore("counter", 0);

store.subscribe(value => {
  console.log("counter →", value);
});

// Functional update — always gets latest value
store.setItem(prev => prev + 1);
store.setItem(0); // reset

localStorage

localStorage Store

Persists until cleared. Cross-tab sync via
native StorageEvent. Type a message.

Usage
import { StorageClient } from "synced-storage/core";

const client = new StorageClient();
const store = client.getOrCreateStore(
  "message", "",
  { strategy: "localStorage" }
);

store.subscribe(value => {
  console.log("message →", value);
});

store.setItem("hello world");
store.removeItem();

sessionStorage + Expiry

sessionStorage Store

Cleared when the tab closes. This store
expires automatically after 15 seconds.

Usage
import { StorageClient } from "synced-storage/core";

const client = new StorageClient();
const store = client.getOrCreateStore(
  "session", "",
  { strategy: "sessionStorage" }
);

store.subscribe(value => {
  console.log("session →", value);
});

store.setItem(`Set at ${new Date().toLocaleTimeString()}`);

// Manual expiry: remove after 15 seconds
setTimeout(() => store.removeItem(), 15_000);
